    // src/lib/nearmap.ts
    export const NEARMAP_BUCKET = "nearmap-images";
    export const NEARMAP_IMAGE_SIZE = "1024x1024";
    export const NEARMAP_BASE = "https://api.nearmap.com";

    export type NearmapCoverage = {
      surveys: Array<{
        id: string;
        transactionToken: string;
        captureDate: string;
      }>;
    };

    /**
     * Transactional path Step 1 — Coverage call.
     * NOTE: Nearmap Transactional API uses LONG,LAT order (opposite of Google Maps).
     * @param lng - longitude (FIRST in Nearmap URL)
     * @param lat - latitude (SECOND in Nearmap URL)
     */
    export function buildCoverageUrl(apiKey: string, lat: number, lng: number): string {
      // Nearmap: LONG,LAT order
      return `${NEARMAP_BASE}/coverage/v2/point/${lng},${lat}?apikey=${encodeURIComponent(apiKey)}&limit=1&resources=Vert`;
    }

    /**
     * Transactional path Step 2 — Image fetch using surveyId + transactionToken.
     * NOTE: point parameter is also LONG,LAT order.
     */
    export function buildImageUrl(
      surveyId: string,
      transactionToken: string,
      lat: number,
      lng: number
    ): string {
      // Nearmap: LONG,LAT order for point param
      return `${NEARMAP_BASE}/staticmap/v3/surveys/${surveyId}/Vert.jpg?point=${lng},${lat}&radius=100&size=${NEARMAP_IMAGE_SIZE}&transactionToken=${encodeURIComponent(transactionToken)}`;
    }

    /**
     * Legacy path — Single-step staticmap.
     * NOTE: center parameter uses LAT,LNG order (opposite of Transactional).
     */
    export function buildLegacyUrl(apiKey: string, lat: number, lng: number): string {
      // Legacy staticmap: LAT,LNG order (different from Transactional)
      return `${NEARMAP_BASE}/staticmap?center=${lat},${lng}&size=${NEARMAP_IMAGE_SIZE}&zoom=21&apikey=${encodeURIComponent(apiKey)}`;
    }

    ```typescript
    // src/app/api/nearmap-image/route.ts
    import { NextResponse } from "next/server";
    import { getSupabaseAdmin } from "@/lib/supabase";
    import {
      NEARMAP_BUCKET,
      buildCoverageUrl,
      buildImageUrl,
      buildLegacyUrl,
      type NearmapCoverage,
    } from "@/lib/nearmap";

    export const runtime = "nodejs";
    export const maxDuration = 30;

    // --- Cache helpers ---
    async function getCachedImageUrl(placeId: string): Promise<string | null> {
      const safeName = `${String(placeId).replace(/[^a-zA-Z0-9_-]/g, "_")}.jpg`;
      const supabase = getSupabaseAdmin();
      const { data: files } = await supabase.storage
        .from(NEARMAP_BUCKET)
        .list("", { search: safeName });
      if (!files?.some((f) => f.name === safeName)) return null;
      const { data } = supabase.storage.from(NEARMAP_BUCKET).getPublicUrl(safeName);
      return data.publicUrl;
    }

    async function uploadImage(placeId: string, jpegBytes: ArrayBuffer): Promise<string> {
      const safeName = `${String(placeId).replace(/[^a-zA-Z0-9_-]/g, "_")}.jpg`;
      const supabase = getSupabaseAdmin();
      const { error } = await supabase.storage.from(NEARMAP_BUCKET).upload(safeName, jpegBytes, {
        contentType: "image/jpeg",
        cacheControl: "31536000",
        upsert: false,
      });
      // Swallow "already exists" — concurrent request race condition, safe to continue
      if (error && !error.message.toLowerCase().includes("already exists")) {
        throw new Error(`Storage upload failed: ${error.message}`);
      }
      const { data } = supabase.storage.from(NEARMAP_BUCKET).getPublicUrl(safeName);
      return data.publicUrl;
    }

    // --- Nearmap fetch ---
    async function fetchNearmapImage(
      apiKey: string,
      lat: number,
      lng: number
    ): Promise<ArrayBuffer> {
      const useLegacy = process.env.NEARMAP_USE_LEGACY === "true";

      if (useLegacy) {
        // Single-step legacy path — LAT,LNG order
        const url = buildLegacyUrl(apiKey, lat, lng);
        const res = await fetch(url, { headers: { Accept: "image/jpeg" } });
        if (!res.ok) throw new Error(`Nearmap legacy fetch failed: ${res.status}`);
        return res.arrayBuffer();
      }

      // Two-step Transactional path
      // Step 1: Coverage call — credits deducted here
      const coverageUrl = buildCoverageUrl(apiKey, lat, lng);
      const coverageRes = await fetch(coverageUrl, { headers: { Accept: "application/json" } });
      if (!coverageRes.ok) throw new Error(`Nearmap coverage call failed: ${coverageRes.status}`);
      const coverage: NearmapCoverage = await coverageRes.json();
      const survey = coverage.surveys?.[0];
      if (!survey?.id) {
        throw new Error("No Nearmap coverage available for this location");
      }

      // Step 2: Image fetch — no additional credit deduction
      const imageUrl = buildImageUrl(survey.id, survey.transactionToken, lat, lng);
      const imageRes = await fetch(imageUrl, { headers: { Accept: "image/jpeg" } });
      if (!imageRes.ok) throw new Error(`Nearmap image fetch failed: ${imageRes.status}`);
      return imageRes.arrayBuffer();
    }

    // --- Route handler ---
    export async function POST(req: Request) {
      const apiKey = process.env.NEARMAP_API_KEY;
      if (!apiKey) {
        return NextResponse.json({ error: "NEARMAP_API_KEY not configured" }, { status: 500 });
      }

      const body = await req.json().catch(() => null);
      const { placeId, lat, lng } = body ?? {};
      if (!placeId || typeof lat !== "number" || typeof lng !== "number") {
        return NextResponse.json({ error: "Missing placeId, lat, or lng" }, { status: 400 });
      }

      try {
        // 1. Cache check — serve cached image immediately if present
        const cached = await getCachedImageUrl(placeId);
        if (cached) return NextResponse.json({ url: cached, cache: "hit" });

        // 2. Fetch from Nearmap — incurs API charge
        const jpegBuffer = await fetchNearmapImage(apiKey, lat, lng);

        // 3. Upload to Supabase Storage
        const url = await uploadImage(placeId, jpegBuffer);

        return NextResponse.json({ url, cache: "miss" });
      } catch (err) {
        console.error("[nearmap-image]", err);
        return NextResponse.json(
          { error: err instanceof Error ? err.message : "Unknown error" },
          { status: 502 }
        );
      }
    }
    ```

<threat_model>
## Trust Boundaries

| Boundary | Description |
|----------|-------------|
| client → POST /api/nearmap-image | Untrusted body: placeId, lat, lng from browser |
| API route → Nearmap API | Server-to-server; API key in query param |
| API route → Supabase Storage | Server-to-server; service role key in header |
| Supabase Storage → browser | Public bucket URL served to `<img>` |

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-01-01 | Information Disclosure | NEARMAP_API_KEY | mitigate | Server-only env var — no NEXT_PUBLIC_ prefix. Route handler validates presence before any fetch. Key never appears in JSON response or client bundle. |
| T-01-02 | Tampering | lat/lng input → Nearmap URL | mitigate | Validate `typeof lat !== "number" \|\| typeof lng !== "number"` returns 400. The lat/lng range check already enforced in preview/page.tsx (lat -44 to -10, lng 112 to 154) prevents SSRF to non-AU coordinates. Route adds its own type check as defence-in-depth. |
| T-01-03 | Tampering | placeId → Supabase Storage path | mitigate | Sanitize with `replace(/[^a-zA-Z0-9_-]/g, "_")` before using as filename — prevents path traversal (e.g., placeId containing `../`). Applied in both getCachedImageUrl and uploadImage. |
| T-01-04 | Elevation of Privilege | Nearmap over-billing via cache bypass | mitigate | Cache check runs unconditionally BEFORE any Nearmap fetch. Concurrent upload race condition handled by catching "already exists" error and returning existing URL — at most one double-charge per race, never unbounded. |
| T-01-05 | Information Disclosure | Supabase public bucket URL enumeration | accept | place_id values are Google-generated opaque strings (e.g., ChIJP3Sa8ziYEmsRUKgyFmh9AQM) — not guessable by brute force for a v1 showcase tool. Acceptable risk. |
| T-01-06 | Denial of Service | Unbounded Nearmap API calls from repeated requests | mitigate | Cache-first pattern means only uncached addresses incur cost. No rate-limiting needed in v1 — single-developer demo tool with low traffic. |
| T-01-07 | Tampering | SUPABASE_SERVICE_ROLE_KEY exposure | mitigate | getSupabaseAdmin() is server-only (no NEXT_PUBLIC_ prefix). Never imported from client components. |
</threat_model>
